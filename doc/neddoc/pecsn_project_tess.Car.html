//
module Car
{
    parameters:
        @display("i=block/cogwheel;bgb=400,400");
        int carIndex;
        int numCars = default(10);  // Define the number of cars

        int minCoordValue @unit(m) = default(0m);// the min coordinate value
        int maxCoordValue @unit(m) = default(500m); // the max coordinate value

        double minSpeed @unit(mps) = default(1mps);
        double maxSpeed @unit(mps) = default(10mps);

        //double coordOffsetFactor = default(20.0);  // Factor to modify the position for each car
        //int randomSpread @unit(m) = default(50m);  // Extra spread for initial positions
    submodules:
        //Custom Random Mobility simple module
        customMobility: MobiltiyModule {
            carId = parent.carIndex;
            @display("p=98,109");
        }
        // Custom Sensing simple Module  
        customSensing: SensingModule {
            carID = parent.carIndex;  // Pass carIndex to customSensing
            @display("p=224,101");
        }
}
